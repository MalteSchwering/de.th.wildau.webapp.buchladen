60      /***
61       * This method must be implemented by sublcasses and : Retrieve the challenge from the captcha Make and return a
62       * clone of the challenge Return the clone It has be design in order to let the service dipose the challenge of the
63       * captcha after rendering. It should be implemented for all captcha type (@see ImageCaptchaService implementations
64       * for exemple)
65       *
66       * @return a Challenge Clone
67       */
68      protected Object getChallengeClone(Captcha captcha) {
69          BufferedImage challenge = (BufferedImage) captcha.getChallenge();
70          BufferedImage clone = new BufferedImage(challenge.getWidth(), challenge.getHeight(), challenge.getType());
71  
72          clone.getGraphics().drawImage(challenge, 0, 0, clone.getWidth(), clone.getHeight(), null);
73          clone.getGraphics().dispose();
74  
75  
76          return clone;
77      }
