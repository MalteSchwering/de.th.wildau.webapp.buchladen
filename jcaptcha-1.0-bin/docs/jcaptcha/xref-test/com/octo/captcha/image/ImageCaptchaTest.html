132     private Captcha unmarshalCaptcha(byte[] marshalledCaptcha) throws IOException, ClassNotFoundException {
133         ObjectInputStream in = new ObjectInputStream(new
134                 ByteArrayInputStream(marshalledCaptcha));
135         Captcha captchaUnserialized = (Captcha) in.readObject();
136         return captchaUnserialized;
137     }
138 
139     private byte[] marshalCaptcha(Captcha captcha) throws IOException {
140         ByteArrayOutputStream arrayOutput = new ByteArrayOutputStream();
141         ObjectOutputStream out = new ObjectOutputStream(arrayOutput);
142         out.writeObject(captcha);
143         out.flush();
144         arrayOutput.close();
145         return arrayOutput.toByteArray();
146     }
