33  public class ComposeDictionaryWordGeneratorTest extends TestCase {
34  
35      protected ComposeDictionaryWordGenerator composeDictionaryWordGenerator;
36      protected String[] wordList = {"1", "12", "123", "123456"};
37      protected int[] lengths = {1, 2, 3, 6};
38  
39      protected String[] badwordList = {"a", "b", "c", "d"};
40      protected int[] badlengths = {1, 1, 1, 1};
41  
42      protected String[] emptywordList = {};
43      protected int emptylength = 10;
44  
45      /***
46       * This method is the setup for each testcase.
47       */
48      protected void setUp() throws Exception {
49          super.setUp();
50          this.composeDictionaryWordGenerator =
51                  new ComposeDictionaryWordGenerator(new ArrayDictionary(wordList));
52      }
53  
54      /***
55       * This testcase verify if the class works normaly.
56       */
57      public void testGetWordIntegerLocale() {
58          for (int i = 0; i < lengths.length; i++) {
59              String test = this.composeDictionaryWordGenerator.getWord(new Integer(this.lengths[i]), Locale.FRENCH);
60              assertNotNull(test);
61              assertTrue(test.length() > 0);
62              assertEquals(lengths[i], test.length());
63          }
64      }
65  
66      /***
67       * This testcase werify if it works with words with one letter in the dictionary.
68       */
69      public void testSmallWordWithOneLetter() {
70          this.composeDictionaryWordGenerator =
71                  new ComposeDictionaryWordGenerator(new ArrayDictionary(this.badwordList));
72          for (int i = 0; i < badlengths.length; i++) {
73              String test = this.composeDictionaryWordGenerator.getWord(new Integer(this.badlengths[i]), Locale.FRENCH);
74              assertNotNull(test);
75              assertTrue(test.length() > 0);
76              assertEquals(badlengths[i], test.length());
77          }
78      }
79  
80      /***
81       * this testcase verify if it works with words with no words in the dictionnary. We verify if the CaptchaException
82       * is correctly trapped.
83       */
84      public void testEmptyDictionnary() {
85          this.composeDictionaryWordGenerator =
86                  new ComposeDictionaryWordGenerator(new ArrayDictionary(this.emptywordList));
87          try {
88              this.composeDictionaryWordGenerator.getWord(new Integer(this.emptylength), Locale.FRENCH);
89              fail("Shouldn't use empty dictionnary");
90          } catch (CaptchaException e) {
91              assertNotNull(e);
92          }
93      }
