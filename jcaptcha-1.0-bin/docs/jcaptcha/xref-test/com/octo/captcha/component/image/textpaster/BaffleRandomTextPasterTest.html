36      protected BaffleRandomTextPaster baffleRandomTextPaster;
37      protected Integer minAcceptedWordLength = new Integer(10);
38      protected Integer maxAcceptedWordLength = new Integer(10);
39      protected Integer numberHoles = new Integer(10);
40      protected Color textColor = Color.black;
41      protected Color holesColor = Color.white;
42  
43      /***
44       * Constructor for BaffleRandomTextPasterTest.
45       */
46      public BaffleRandomTextPasterTest(String name) {
47          super(name);
48      }

50      protected void setUp() throws Exception {
51          super.setUp();
52          this.baffleRandomTextPaster = new BaffleRandomTextPaster(this.minAcceptedWordLength,
53                  this.maxAcceptedWordLength,
54                  this.textColor,
55                  this.numberHoles,
56                  this.holesColor);
57      }
58  
59      public void testPasteText() {
60          BufferedImage imageTest = new BufferedImage(100, 50, BufferedImage.TYPE_INT_RGB);
61          AttributedString stringTest = new AttributedString("test");
62          BufferedImage test = null;
63          try {
64              test = this.baffleRandomTextPaster.pasteText(imageTest, stringTest);
65              assertNotNull(test);
66              assertEquals(imageTest.getHeight(), test.getHeight());
67              assertEquals(imageTest.getWidth(), test.getWidth());
68          } catch (CaptchaException e) {
69              assertNotNull(e);
70          }
71      }
72  
73      public void testGetMaxAcceptedWordLenght() {

76      }
