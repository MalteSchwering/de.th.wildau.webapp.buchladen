38  public class FilteredDoubleRandomListGimpyEngine

41      protected void buildInitialFactories() {
42  
43          com.jhlabs.image.RippleFilter rippleBack = new com.jhlabs.image.RippleFilter();
44  
45          rippleBack.setWaveType(com.jhlabs.image.RippleFilter.NOISE);
46          rippleBack.setXAmplitude(5);
47          rippleBack.setYAmplitude(5);
48          rippleBack.setXWavelength(10);
49          rippleBack.setYWavelength(10);
50          rippleBack.setEdgeAction(com.jhlabs.image.TransformFilter.CLAMP);
51  
52          TextPaster paster =
53                  new DoubleRandomTextPaster(new Integer(8), new Integer(15),
54                          Color.black);
55          BackgroundGenerator back = new MultipleShapeBackgroundGenerator(
56                  new Integer(200), new Integer(100));
57          FontGenerator font =
58                  new DeformedRandomFontGenerator(new Integer(25),
59                          new Integer(27));
60          com.octo.captcha.component.word.wordgenerator.WordGenerator words =
61                  new com.octo.captcha.component.word.wordgenerator.DictionaryWordGenerator(
62                          new com.octo.captcha.component.word.FileDictionary(
63                                  "toddlist"));
64  
65          com.octo.captcha.component.image.wordtoimage.WordToImage word2image =
66                  new com.octo.captcha.component.image.wordtoimage.FilteredComposedWordToImage(
67                          font,
68                          back,
69                          paster,
70                          new ImageFilter[]{rippleBack},
71                          new ImageFilter[]{},
72                          new ImageFilter[]{});
73          com.octo.captcha.image.ImageCaptchaFactory factory = new com.octo.captcha.image.gimpy.GimpyFactory(
74                  words, word2image);
75          this.addFactory(factory);
76      }
