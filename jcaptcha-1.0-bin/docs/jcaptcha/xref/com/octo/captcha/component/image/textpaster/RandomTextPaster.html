52      /***
53       * Paste the text randomly on the background.
54       * <p/>
55       * Pastes the attributed string on the backround image and return the final image. Implementation must take into
56       * account the fact that the text must be readable by human and non by programs. <p/>
57       *
58       * @return the final image
59       *
60       * @throws com.octo.captcha.CaptchaException
61       *          if any exception accurs during paste routine.
62       */
63      public BufferedImage pasteText(final BufferedImage background,
64                                     final AttributedString attributedString) {
65          BufferedImage out = copyBackground(background);
66          Graphics2D g2 = pasteBackgroundAndSetTextColor(out, background);
67          g2.setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS,
68                  RenderingHints.VALUE_FRACTIONALMETRICS_ON);
69          g2.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_QUALITY);
70          g2.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
71                  RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
72          // this doesn't do anything in JDK 1.4, but maybe it will in JDK 1.5
73          // attributedString.addAttribute(TextAttribute.WIDTH, TextAttribute.WIDTH_EXTENDED);
74  
75          // convert string into a series of glyphs we can work with
76          ChangeableAttributedString newAttrString = new ChangeableAttributedString(g2,
77                  attributedString, kerning);
78  
79          // space out the glyphs with a little kerning
80          newAttrString.useMinimumSpacing(kerning);
81          // shift string to a random spot in the output imge
82          newAttrString.moveToRandomSpot(background);
83          // now draw each glyph at the appropriate spot on the image.
84          if (isManageColorPerGlyph())
85              newAttrString.drawString(g2, getColorGenerator());
86          else
87              newAttrString.drawString(g2);
88  
89          g2.dispose();
90          return out;
91      }
