55      /***
56       * Pastes the attributed string on the backround image and return the final image. Implementation must take into
57       * account the fact that the text must be readable by human and non by programs. Pastes two times the attributed
58       * string with a random decalage from width/20 and height/2
59       *
60       * @return the final image
61       */
62      public BufferedImage pasteText(final BufferedImage background,
63                                     final AttributedString attributedWord) {
64          int x = (background.getWidth()) / 20;
65          int y = (background.getHeight()) / 2;
66          BufferedImage out = copyBackground(background);
67          Graphics2D g2 = pasteBackgroundAndSetTextColor(out, background);
68  
69          //        g2.drawString(attributedWord.getIterator(), x, y);
70          //        g2.drawString(attributedWord.getIterator(), x +
71          //                myRandom.nextInt(5) + 5, y + myRandom.nextInt(5) + 5);
72          //        g2.dispose();
73  
74          // convert string into a series of glyphs we can work with
75          ChangeableAttributedString newAttrString = new ChangeableAttributedString(g2,
76                  attributedWord, 0);
77  
78          // space out the glyphs with a little kerning
79          newAttrString.useMinimumSpacing(0);
80          newAttrString.useMonospacing(0);
81          // shift string to a random spot in the output imge
82          newAttrString.moveTo(x, y);
83          // now draw each glyph at the appropriate spot on the image.
84          if (isManageColorPerGlyph()) {
85              newAttrString.drawString(g2, getColorGenerator());
86          } else {
87              newAttrString.drawString(g2);
88          }
89  
90          newAttrString.moveTo(x + myRandom.nextInt(5) + 5, y + myRandom.nextInt(5) + 5);
91  
92          //draw again
93          if (isManageColorPerGlyph()) {
94              newAttrString.drawString(g2, getColorGenerator());
95          } else {
96              newAttrString.drawString(g2);
97          }
98          return out;
99      }
