26  /***
27   * ImageCaptchaFilter is a J2EE Filter designed to add image captchas to the entry forms of existing MVC web
28   * applications. <br> Current features of ImageCaptchaFilter are: <ul> <li> Generation and rendering of Captcha images
29   * (as JPEG) at runtime for inclusion in an existing entry form. ImageCaptchaFilter use a ImageCaptchaEngine to generate
30   * captchas : a simple one (com.octo.captcha.image.gimpy.MultipleGimpyEngine, which displays a random string composed
31   * with characters A,B,C,D and E) is provided with jcaptcha-j2ee, but you can code your own or use one of those provided
32   * with jcaptcha-sample ; </li> <li> Verification of the HTTP client entry in response to the challenge displayed as an
33   * image in the entry form: Redirection of the HTTP client to an error page if the captcha challenge is not passed ;
34   * Transparent follow up of the request to the web application, without any information concerning the captcha, if the
35   * challenge is successfully passed ; Many forms can be protected by captchas in the same web application, each one
36   * having its own error redirection page ; </li> </ul> The JMX Management interface for ImageCaptchaFilter is
37   * PixCaptchaFilterMbean.
38   *
39   * @author <a href="mailto:sbr@octo.com">Sebastien Brunot</a>
40   * @author <a href="mailto:mag@jcaptcha.net">Marc-Antoine GARRIGUE</a>
41   * @version $Id: ImageCaptchaFilter.java 466 2007-12-27 09:38:03Z antoineveret $
42   */

203     /***
204      * @see javax.servlet.Filter#init(javax.servlet.FilterConfig)
205      */
206     public void init(final FilterConfig theFilterConfig)
207             throws ServletException {
208 
209         // get rendering URL from web.xml
210         this.captchaRenderingURL =
211                 FilterConfigUtils.getStringInitParameter(theFilterConfig,
212                         CAPTCHA_RENDERING_URL_PARAMETER,
213                         true);
214 
215         // get rendering URL from web.xml
216         captchaErrorURL =
217                 FilterConfigUtils.getStringInitParameter(theFilterConfig,
218                         CAPTCHA_ERROR_URL_PARAMETER,
219                         true);
220 
221         // get verification URLs from web.xml (CSV list of URLs)
222         String captchaVerificationURLs =
223                 FilterConfigUtils.getStringInitParameter(theFilterConfig,
224                         CAPTCHA_VERIFICATION_URLS_PARAMETER,
225                         true);
226 
227         // get forward error URLs from web.xml (CSV list of URLs)
228         String captchaForwardErrorURLs =
229                 FilterConfigUtils.getStringInitParameter(theFilterConfig,
230                         CAPTCHA_FAIL_URLS_PARAMETER,
231                         true);
232 
233         // initialize the verificationForwards hashtable
234         StringTokenizer verificationURLs =
235                 new StringTokenizer(captchaVerificationURLs, CSV_DELIMITER, false);
236         StringTokenizer forwardErrorURLs =
237                 new StringTokenizer(captchaForwardErrorURLs, CSV_DELIMITER, false);
238         if (verificationURLs.countTokens() != forwardErrorURLs.countTokens()) {
239             // The URL lists are not consistant (there should be a forward and
240             // a success for each verification URL)
241             throw new ServletException(CAPTCHA_VERIFICATION_URLS_PARAMETER
242                     + " and "
243                     + CAPTCHA_FAIL_URLS_PARAMETER
244                     + " values are not consistant in web.xml : there should be"
245                     + " exactly one forward error for each verification URL !");
246         }
247         while (verificationURLs.hasMoreTokens()) {
248             // Create a ForwardInfo for each verification URL and store it in
249             // the verificationForward hashtable
250             this.verificationForwards.put(verificationURLs.nextToken(),
251                     forwardErrorURLs.nextToken());
252         }
253         //   get captcha Question parameter name from web.xml
254         this.captchaQuestionParameterName =
255                 FilterConfigUtils.getStringInitParameter(theFilterConfig,
256                         CAPTCHA_QUESTION_NAME_PARAMETER, true);
257 
258         // get captcha ID parameter name from web.xml
259 //        this.captchaIDParameterName =
260 //                FilterConfigUtils.getStringInitParameter(theFilterConfig,
261 //                        CAPTCHA_ID_PARAMETER_NAME_PARAMETER,
262 //                        true);
263 
264         // get challenge response parameter name from web.xml
265         this.captchaChallengeResponseParameterName =
266                 FilterConfigUtils.getStringInitParameter(theFilterConfig,
267                         CAPTCHA_RESPONSE_PARAMETER_NAME_PARAMETER,
268                         true);
269 
270         // get service parameter class name from web.xml
271         this.captchaServiceClassName =
272                 FilterConfigUtils.getStringInitParameter(theFilterConfig,
273                         CAPTCHA_SERVICE_CLASS_PARAMETER,
274                         true);
275 
276         // get from web.xml the indicator signaling if the CaptchaFilter
277         // should be registered to an MBean Server
278         this.captchaRegisterToMBeanServer =
279                 FilterConfigUtils.getBooleanInitParameter(theFilterConfig,
280                         CAPTCHA_REGISTER_TO_MBEAN_SERVER_PARAMETER,
281                         false);
282 
283         // create the ImageCaptchaService
284         try {
285             this.captchaService = (ImageCaptchaService) Class.forName(captchaServiceClassName).newInstance();
286         } catch (InstantiationException e) {
287             //log.debug("message", e);
288 
289             throw new CaptchaServiceException(e);
290         } catch (IllegalAccessException e) {
291             //log.debug("message", e);
292 
293             throw new CaptchaServiceException(e);
294         } catch (ClassNotFoundException e) {
295             //log.debug("message", e);
296             throw new CaptchaServiceException(e);
297         }
298         // register the ImageCaptchaService to an MBean server if specified
299         if (this.captchaRegisterToMBeanServer && captchaService instanceof ManageableCaptchaService) {
300             ManageableCaptchaService manageable = (ManageableCaptchaService) captchaService;
301             JMXRegistrationHelper.registerToMBeanServer(manageable, JMX_REGISTERING_NAME);
302         }

307     /***
308      * @see javax.servlet.Filter#doFilter(javax.servlet.ServletRequest, javax.servlet.ServletResponse,
309      *      javax.servlet.FilterChain)
310      */
311     public void doFilter(final ServletRequest theRequest,
312                          final ServletResponse theResponse,
313                          final FilterChain theFilterChain)
314             throws IOException, ServletException {
315         HttpServletRequest request = (HttpServletRequest) theRequest;
316         HttpServletResponse response = (HttpServletResponse) theResponse;
317 
318         // Get the URL the user asked for
319         StringBuffer servletPathBuff = new StringBuffer()
320                 .append(request.getServletPath());
321         if (request.getQueryString() != null) servletPathBuff.append("?").append(request.getQueryString());
322         String servletPathInfo = servletPathBuff.toString();
323         if (servletPathInfo.startsWith(this.captchaRenderingURL)) {
324             // This is the URL used to ask for captcha generation : do it !
325             try {
326                 this.generateAndRenderCaptcha(request, response);
327             } catch (Throwable e) {
328                 // Redirect to the error URL
329 
330                 response.sendRedirect(captchaErrorURL);
331                 e.printStackTrace();
332             }
333         } else if (this.verificationForwards.containsKey(servletPathInfo)) {
334             // This is the URL used to ask for captcha challenge verification :
335             // do it !
336             this.verifyAnswerToACaptchaChallenge(request,
337                     response,
338                     servletPathInfo,
339                     theFilterChain);
340         } else {
341             // this is a question : add it and follow to the filter chain...
342             addQuestionToRequest(request, response);
343             theFilterChain.doFilter(theRequest, theResponse);
344         }
345     }

371     /***
372      * Generate a new ImageCaptcha, store it in the internal store and render it as JPEG to the client. Captcha are
373      * localized using request locale ID used are session ID This method returns a 404 to the client instead of the
374      * image if the request isn't correct (missing parameters, etc...).
375      *
376      * @param theRequest  the request
377      * @param theResponse the response
378      *
379      * @throws java.io.IOException if a problem occurs during the jpeg generation process
380      */
381     private void generateAndRenderCaptcha(HttpServletRequest theRequest,
382                                           HttpServletResponse theResponse)
383             throws IOException {
384 
385 
386         String captchaID = theRequest.getSession().getId();//(String) theRequest.getParameter(captchaIDParameterName);
387 
388         // call the ManageableImageCaptchaService methods
389         byte[] captchaChallengeAsJpeg = null;
390         ByteArrayOutputStream jpegOutputStream = new ByteArrayOutputStream();
391         try {
392             BufferedImage challenge =
393                     this.captchaService.getImageChallengeForID(captchaID, theRequest.getLocale());
394             // the output stream to render the captcha image as jpeg into
395 
396             // a jpeg encoder
397             JPEGImageEncoder jpegEncoder =
398                     JPEGCodec.createJPEGEncoder(jpegOutputStream);
399             jpegEncoder.encode(challenge);
400         } catch (IllegalArgumentException e) {
401             // log a security warning and return a 404...
402 //            if (log.isWarnEnabled())
403 //            {
404 //                log.warn(
405 //                    "There was a try from "
406 //                        + theRequest.getRemoteAddr()
407 //                        + " to render an URL without ID"
408 //                        + " or with a too long one");
409 //                theResponse.sendError(HttpServletResponse.SC_NOT_FOUND);
410             return;
411 //            }
412         } catch (CaptchaServiceException e) {
413             // log and return a 404 instead of an image...
414 //            log.warn(
415 //                "Error trying to generate a captcha and "
416 //                    + "render its challenge as JPEG",
417 //                e);
418             theResponse.sendError(HttpServletResponse.SC_NOT_FOUND);
419             return;
420         }
421 
422         captchaChallengeAsJpeg = jpegOutputStream.toByteArray();
423 
424         // render the captcha challenge as a JPEG image in the response
425         theResponse.setHeader("Cache-Control", "no-store");
426         theResponse.setHeader("Pragma", "no-cache");
427         theResponse.setDateHeader("Expires", 0);
428 
429         theResponse.setContentType("image/jpeg");
430         ServletOutputStream responseOutputStream =
431                 theResponse.getOutputStream();
432         responseOutputStream.write(captchaChallengeAsJpeg);

436     /***
437      * Verify client answer to a captcha challenge, and forward to the correponding success or error URL. This method
438      * returns a 404 to the client instead of forwarding if the request isn't correct (missing parameters, etc...).
439      *
440      * @param theRequest         the request
441      * @param theResponse        the response
442      * @param theVerificationURL the verification URL that was called
443      *
444      * @throws java.io.IOException            if the image tranformation fails
445      * @throws javax.servlet.ServletException if a servlet exception occurs
446      */
447     private void verifyAnswerToACaptchaChallenge(HttpServletRequest theRequest,
448                                                  HttpServletResponse theResponse,
449                                                  String theVerificationURL,
450                                                  FilterChain theFilterChain)
451             throws IOException, ServletException {
452         // get captcha ID from the session!!
453         String captchaID = theRequest.getSession().getId();//theRequest.getParameter(captchaIDParameterName);
454 
455         // get challenge response from the request
456         String challengeResponse =
457                 theRequest.getParameter(captchaChallengeResponseParameterName);
458         if (challengeResponse == null) {
459             // If the challenge response is not specified, forward error
460             this.redirectError(theVerificationURL, theRequest, theResponse);
461             return;
462         }
463 
464         // Call the ManageableImageCaptchaService method
465         Boolean isResponseCorrect = Boolean.FALSE;
466         try {
467             isResponseCorrect =
468                     this.captchaService.validateResponseForID(captchaID,
469                             challengeResponse);
470         } catch (CaptchaServiceException e) {
471             // nothing to do : isResponseCorrect is false
472             // so the user will be redirected to the error page
473         }
474 
475         // forward user to the success URL or redirect it to the error URL
476         if (isResponseCorrect.booleanValue()) {
477             // clean the request and call the next element in filter chain
478             // (forward success)
479             this.forwardSuccess(theFilterChain, theRequest, theResponse);
480         } else {
481             // forward
482             this.redirectError(theVerificationURL, theRequest, theResponse);
483         }
484     }

486     /***
487      * Redirect request to the Error URL
488      *
489      * @param theVerificationURL the verification URL for which there is an error redirect
490      * @param theRequest         the request
491      * @param theResponse        the response
492      *
493      * @see javax.servlet.RequestDispatcher#forward(javax.servlet.ServletRequest, javax.servlet.ServletResponse)
494      */
495     private void redirectError(String theVerificationURL,
496                                HttpServletRequest theRequest,
497                                HttpServletResponse theResponse)
498             throws ServletException {
499         this.removeParametersFromRequest(theRequest);
500         try {
501             // Redirect to the error URL
502             String forwardErrorURL =
503                     theRequest.getContextPath()
504                             + (String) this.verificationForwards.get(theVerificationURL);
505             theResponse.sendRedirect(forwardErrorURL);
506         } catch (IOException e) {
507             throw new ServletException(e);
508         }
509     }
