7   package com.octo.captcha;

9   import java.io.Serializable;
10  
11  /***
12   * This interface represent a CAPTCHA.
13   * <p/>
14   * A CAPTCHA is a program that can generate and grade tests that: <ul> <li>Most humans can pass.</li> <li>Current
15   * computer programs can't pass</li> </ul> see http://www.captcha.net/ for sample, articles, and definitions.
16   * <p/>
17   * A captcha is basically a test composed of : <ul> <li>A question about :</li> <li>a challenge (can be an image for
18   * image captchas or a sound, or wathever)</li> <li>a validation routine a of a given response</li> </ul>
19   * <p/>
20   * This is a container for the CAPTCHA challenge which is also able to validate the answer. Class implementing this
21   * interface must follow the rules : <ul> <li>As all 'components' of this project, all contructed objects should be
22   * abble to word (ie no need to set other properties to obtain a fully fonctional object). constructor</li> <li>It must
23   * not build the challenge! use instead the CaptchaFactory</li> <li>It must know how to validate the answer</li> <li>It
24   * must not expose the answer</li> <li>It must dispose the challenge when the getChallenge method is called(The
25   * challenge must be showed only once ) </li> </ul> ;
26   *
27   * @author <a href="mailto:mag@jcaptcha.net">Marc-Antoine Garrigue</a>
28   * @version 1.0
29   */
30  public interface Captcha extends Serializable {
31  
32      /***
33       * Accessor captcha question.
34       *
35       * @return the question
36       */
37      String getQuestion();
38  
39      /***
40       * Accerssor for the questionned challenge.
41       *
42       * @return the challenge (may be an image for image captcha...
43       */
44      Object getChallenge();
45  
46      /***
47       * Validation routine for the response.
48       *
49       * @param response to the question concerning the chalenge
50       *
51       * @return true if the answer is correct, false otherwise.
52       */
53      Boolean validateResponse(Object response);
54  
55      /***
56       * Dispose the challenge, once this method is call the getChallenge method will return null.<br> It has been added
57       * for technical reasons : a captcha is always used in a two step fashion<br> First submit the challenge, and then
58       * wait until the response arrives.<br> It had been asked to have a method to dispose the challenge that is no
59       * longer used after being dipslayed. So here it is!
60       */
61      void disposeChallenge();
62  
63      /***
64       * This method should return true if the getChalenge method has been called (has been added in order to properly
65       * manage the captcha state.
66       *
67       * @return true if getChallenge has been called false otherwise.
68       */
69      Boolean hasGetChalengeBeenCalled();
70  
71  }
